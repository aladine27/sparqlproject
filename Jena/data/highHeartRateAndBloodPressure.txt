PREFIX ns: <http://www.semanticweb.org/health-tracker#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?individual ?heartRate ?bloodPressure
WHERE {
  ?individual rdf:type ns:User .
  ?individual ns:heartRate ?heartRate .
  ?individual ns:bloodPressure ?bloodPressure .
  FILTER(?heartRate > 75 && regex(?bloodPressure, "^12[0-9]/[6-9][0-9]$|^13[0-9]/[6-9][0-9]$"))
}
